    # Se n√£o existe, insere um novo com o tipo de neg√≥cio
    response = supabase_client.table('mercados_monitorados').insert({
        'user_id': user_id, 
        'termo': termo, 
        'localizacao': localizacao,
        'tipo_negocio': tipo_negocio
    }).execute()
    return response.data[0]['id']
